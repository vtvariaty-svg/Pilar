function CTAButton({ children, message, variant = "dark" }) {

    variant === "light"
      ? "bg-white text-neutral-950 hover:bg-neutral-100"
      : "bg-neutral-950 text-white hover:bg-neutral-800";

    <a
      href={whatsappLink(message)}
      target="_blank"

}
